#create a vector of all of the integers from 1 to 100 that are not divisible by 2, 3, or 7.

vec5 <- seq(1:100) #made the vector

vec5 <- vec5[!(vec5 %%2 == 0 | vec5 %%3 == 0 | vec5 %%7 == 0)] #in order to make sure it was numbers NOT divisible have to put the parameter in () and have ! ahead of that within the [] so the ! will apply to each term separated by | for or
